### Class Component 
```javascript
import { Component } from 'react'

class Hello extends Component {
    constructor() {
        super()
        this.state = {
            count: 0
        }
    }
    
    handleClick = () => {
        this.setState({
            count: this.state.count + 1
        })
    }


    render() {
        const {count} = this.state
        return (
            // JSX
            <div>
                <button onClick={this.handleClick}></button>
                <p>Clicked {count} times</p>
            </div>
        )
    }
}
```
